# Fix HDMI w/ Argon case (https://forum.libreelec.tv/thread/22079-rpi4-no-hdmi-sound-using-argon-one-case/):

hdmi_group=1
hdmi_mode=16

disable_overscan=1

# Enable IR configuration

# Switch TV to Game Mode when possible
edid_content_type=4

# Fix lag on cheap mice
usbhid.mousepoll=0

# Boot using custom kernel
#kernel=kernel-retrokit.img

[pi4]
# Overclocking
over_voltage=2
arm_freq=1750

[all]

# Disable wifi (here to avoid conflicts with the other dtoverlay)
dtoverlay=disable-wifi
